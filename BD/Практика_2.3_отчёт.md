## Задание 1 — Автоснижение стоимости объектов по давности и средней оценке

```sql
CREATE OR REPLACE FUNCTION apply_property_discounts()
RETURNS INTEGER
LANGUAGE plpgsql
AS $$
DECLARE
  updated_count INTEGER := 0;
BEGIN
  WITH rated AS (
    SELECT
      p.id,
      p.created_at::DATE AS created_date,
      p.price_rub,
      AVG(r.rating)::NUMERIC AS avg_rating
    FROM properties p
    LEFT JOIN property_ratings r ON r.property_id = p.id
    GROUP BY p.id, p.created_at, p.price_rub
  ),
  to_update AS (
    SELECT
      id,
      CASE
        WHEN created_date <= (CURRENT_DATE - INTERVAL '12 months') AND avg_rating < 4 THEN 0.20
        WHEN created_date <= (CURRENT_DATE - INTERVAL '9 months')  AND avg_rating < 5 THEN 0.10
        WHEN created_date <= (CURRENT_DATE - INTERVAL '6 months')  AND avg_rating < 6 THEN 0.05
        ELSE 0
      END AS discount
    FROM rated
    WHERE avg_rating IS NOT NULL
  )
  UPDATE properties p
    SET price_rub = ROUND(p.price_rub * (1 - u.discount), 2)
  FROM to_update u
  WHERE p.id = u.id
    AND u.discount > 0;

  GET DIAGNOSTICS updated_count = ROW_COUNT;
  RETURN updated_count;
END;
$$;
```

## Задание 2 — Проверка: дата продажи должна быть позже даты размещения объявления

```sql
CREATE OR REPLACE FUNCTION find_sales_date_anomalies()
RETURNS TABLE(result TEXT)
LANGUAGE plpgsql
AS $$
DECLARE
  rec RECORD;
  any_found BOOLEAN := FALSE;
BEGIN
  FOR rec IN
    SELECT
      p.address
    FROM sales s
    JOIN properties p ON p.id = s.property_id
    WHERE s.sale_date::DATE < p.created_at::DATE
    ORDER BY p.address
  LOOP
    any_found := TRUE;
    result := rec.address;
    RETURN NEXT;
  END LOOP;

  IF NOT any_found THEN
    result := 'Аномалий не найдено';
    RETURN NEXT;
  END IF;
END;
$$;
```

## Задание 3 — Проверка: общая площадь = сумма площадей комнат

```sql
CREATE OR REPLACE FUNCTION check_property_area_consistency()
RETURNS TABLE(address TEXT, difference_m2 NUMERIC, status TEXT)
LANGUAGE plpgsql
AS $$
DECLARE
  rec RECORD;
  any_bad BOOLEAN := FALSE;
BEGIN
  FOR rec IN
    SELECT
      p.address,
      (p.total_area_m2 - COALESCE(SUM(pr.room_area_m2), 0))::NUMERIC AS diff
    FROM properties p
    LEFT JOIN property_rooms pr ON pr.property_id = p.id
    GROUP BY p.id, p.address, p.total_area_m2
    HAVING ABS((p.total_area_m2 - COALESCE(SUM(pr.room_area_m2), 0))::NUMERIC) > 0.01
    ORDER BY p.address
  LOOP
    any_bad := TRUE;
    address := rec.address;
    difference_m2 := rec.diff;
    status := 'НЕСООТВЕТСТВИЕ';
    RETURN NEXT;
  END LOOP;

  IF NOT any_bad THEN
    address := NULL;
    difference_m2 := 0;
    status := 'ВСЕ ПЛОЩАДИ СООТВЕТСТВУЮТ';
    RETURN NEXT;
  END IF;
END;
$$;
```

## Задание 4 — Расчёт комиссии риэлтора и запись в таблицу продаж

```sql
CREATE OR REPLACE FUNCTION recalc_sales_commission()
RETURNS INTEGER
LANGUAGE plpgsql
AS $$
DECLARE
  updated_count INTEGER := 0;
BEGIN
  UPDATE sales s
    SET commission_rub = ROUND(
      s.sale_price_rub * CASE
        WHEN s.sale_price_rub < 1000000 THEN 0.02
        WHEN s.sale_price_rub < 3000000 THEN 0.019
        ELSE 0.017
      END
    , 2);

  GET DIAGNOSTICS updated_count = ROW_COUNT;
  RETURN updated_count;
END;
$$;
```
