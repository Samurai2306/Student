## Задание 1 (практика 3.1) — Предупреждение при разнице заявленной и продажной стоимости более 20%

**Условие задачи.** Создать триггер, который при продаже будет выводить предупреждающее сообщение при разнице заявленной и продажной стоимости объекта недвижимости **более чем на 20%**.

*Реализация в отчёте:* заявленная стоимость — `properties.price_rub`, фактическая цена продажи — `sales.sale_price_rub`; сравнение через относительное отклонение от заявленной цены, порог 20%.

```sql
CREATE OR REPLACE FUNCTION tf_sales_warn_price_vs_listing()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
  list_price NUMERIC(14, 2);
  rel_diff NUMERIC;
BEGIN
  SELECT p.price_rub
    INTO list_price
    FROM properties p
    WHERE p.id = NEW.property_id;

  IF list_price IS NULL OR list_price = 0 THEN
    RETURN NEW;
  END IF;

  rel_diff := ABS(NEW.sale_price_rub - list_price) / list_price;
  IF rel_diff > 0.20 THEN
    RAISE WARNING
      'Продажа объекта %: разница заявленной стоимости (%) и продажной (%) составляет более 20%% (относительно заявленной: %).',
      NEW.property_id,
      list_price,
      NEW.sale_price_rub,
      ROUND(100 * rel_diff, 2);
  END IF;

  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_sales_warn_price_vs_listing
BEFORE INSERT OR UPDATE ON sales
FOR EACH ROW
EXECUTE PROCEDURE tf_sales_warn_price_vs_listing();
```

## Задание 2 (практика 3.1) — Запрет повторной продажи одного объекта

**Условие задачи.** Создать триггер, который при добавлении новой продажи будет осуществлять проверку на повторную продажу объекта недвижимости. Если в таблице «Продажи» уже имеется запись о продаже данного объекта, выводить соответствующее сообщение и **запрещать** новую продажу.

*Реализация в отчёте:* триггер на `BEFORE INSERT OR UPDATE` по таблице `sales`; при обнаружении другой строки с тем же `property_id` — `RAISE EXCEPTION`.

```sql
CREATE OR REPLACE FUNCTION tf_sales_block_repeat_property()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
  IF EXISTS (
    SELECT 1
    FROM sales s
    WHERE s.property_id = NEW.property_id
      AND s.id IS DISTINCT FROM NEW.id
  ) THEN
    RAISE EXCEPTION
      'Объект недвижимости с id=% уже имеет запись о продаже. Повторная продажа запрещена.',
      NEW.property_id
      USING ERRCODE = '23514';
  END IF;

  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_sales_block_repeat_property
BEFORE INSERT OR UPDATE ON sales
FOR EACH ROW
EXECUTE PROCEDURE tf_sales_block_repeat_property();
```

## Задание 3 (практика 3.1) — Превышение суммы площадей комнат над общей площадью объекта

**Условие задачи.** Создать триггер, который при добавлении новой записи в таблицу «Структура объекта недвижимости» проверяет несоответствие суммы площадей всех заявленных комнат (**в большую сторону**) с общей площадью объекта недвижимости. Выводить сообщение, **на сколько превышена площадь**.

*Реализация в отчёте:* таблица структуры комнат — `property_rooms`; общая площадь объекта — `properties.total_area_m2`; при превышении суммы комнат — `RAISE WARNING` с величиной превышения в м².

```sql
CREATE OR REPLACE FUNCTION tf_property_rooms_warn_room_sum()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
  total_rooms NUMERIC(12, 2);
  total_prop NUMERIC(12, 2);
  excess NUMERIC(12, 2);
BEGIN
  SELECT COALESCE(SUM(pr.room_area_m2), 0) + COALESCE(NEW.room_area_m2, 0)
    INTO total_rooms
    FROM property_rooms pr
    WHERE pr.property_id = NEW.property_id;

  SELECT p.total_area_m2
    INTO total_prop
    FROM properties p
    WHERE p.id = NEW.property_id;

  IF total_prop IS NULL THEN
    RETURN NEW;
  END IF;

  IF total_rooms > total_prop + 0.01 THEN
    excess := total_rooms - total_prop;
    RAISE WARNING
      'Объект %: сумма площадей комнат (%) м² превышает общую площадь объекта (%) м² на % м².',
      NEW.property_id,
      ROUND(total_rooms, 2),
      ROUND(total_prop, 2),
      ROUND(excess, 2);
  END IF;

  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_property_rooms_warn_sum
BEFORE INSERT ON property_rooms
FOR EACH ROW
EXECUTE PROCEDURE tf_property_rooms_warn_room_sum();
```

*Реализация в отчёте:* дата размещения — `properties.created_at`, дата продажи — `sales.sale_date`; сравнение по календарным дням; предупреждение — `RAISE WARNING`.

```sql
CREATE OR REPLACE FUNCTION tf_sales_warn_sale_before_listing()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
  listed_at TIMESTAMP;
BEGIN
  SELECT p.created_at
    INTO listed_at
    FROM properties p
    WHERE p.id = NEW.property_id;

  IF listed_at IS NOT NULL AND NEW.sale_date::DATE < listed_at::DATE THEN
    RAISE WARNING
      'Продажа объекта %: дата продажи (%) раньше даты размещения объявления (%).',
      NEW.property_id,
      NEW.sale_date::DATE,
      listed_at::DATE;
  END IF;

  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_sales_warn_sale_before_listing
BEFORE INSERT OR UPDATE ON sales
FOR EACH ROW
EXECUTE PROCEDURE tf_sales_warn_sale_before_listing();
```

## Задание 5 (практика 3.1) — Таблица «Журнал» и логирование операций с «Продажами»

**Условие задачи.** Создать таблицу «Журнал»: **дата и время операции**, **операция** (`INSERT`, `UPDATE`, `DELETE`), **пользователь**. Создать триггер, который будет формировать новую запись в этой таблице при добавлении / изменении / удалении строки в таблице «Продажи».

*Реализация в отчёте:* таблица `sales_journal` (время `operated_at`, тип операции `operation`, пользователь `db_user` ← `CURRENT_USER`); триггер `AFTER INSERT OR UPDATE OR DELETE` на `sales`.

```sql
CREATE OR REPLACE FUNCTION tf_sales_journal_aiud()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
  IF TG_OP = 'INSERT' THEN
    INSERT INTO sales_journal (operation, db_user, sale_id)
    VALUES ('INSERT', CURRENT_USER, NEW.id);
  ELSIF TG_OP = 'UPDATE' THEN
    INSERT INTO sales_journal (operation, db_user, sale_id)
    VALUES ('UPDATE', CURRENT_USER, NEW.id);
  ELSIF TG_OP = 'DELETE' THEN
    INSERT INTO sales_journal (operation, db_user, sale_id)
    VALUES ('DELETE', CURRENT_USER, OLD.id);
  END IF;

  RETURN COALESCE(NEW, OLD);
END;
$$;

CREATE TRIGGER trg_sales_journal_aiud
AFTER INSERT OR UPDATE OR DELETE ON sales
FOR EACH ROW
EXECUTE PROCEDURE tf_sales_journal_aiud();
```

## Задание 1 (практика 3.2) — Таблица «Бонусы» и пересчёт 5% при добавлении / удалении продажи

**Условие задачи.** Добавить таблицу «Бонусы», в которой будет **две колонки**: код риэлтора и размер накопленных бонусов. Размер бонуса рассчитывается по формуле **стоимость продажи × 5%**. Создать триггер, который будет автоматически увеличивать размер накопленного бонуса при добавлении новой продажи. Необходимо учесть, что продажа риэлтора может быть **впервые**, следовательно необходимо **добавить новую запись** в таблицу. При **удалении** данных о продаже размер накопленного бонуса должен **уменьшиться**.

*Реализация в отчёте:* таблица `realtor_bonuses` (`realtor_id`, `bonus_amount`); код риэлтора — первичный ключ, ссылается на `realtors.id`; начисление — `AFTER INSERT` на `sales`; уменьшение — `AFTER DELETE`; первая продажа — `INSERT ... ON CONFLICT (realtor_id) DO UPDATE`.

```sql
CREATE OR REPLACE FUNCTION tf_sales_bonus_ai()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
  add_amt NUMERIC(14, 2);
BEGIN
  add_amt := ROUND(NEW.sale_price_rub * 0.05, 2);

  INSERT INTO realtor_bonuses (realtor_id, bonus_amount)
  VALUES (NEW.realtor_id, add_amt)
  ON CONFLICT (realtor_id) DO UPDATE
  SET bonus_amount = realtor_bonuses.bonus_amount + EXCLUDED.bonus_amount;

  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_sales_bonus_ai
AFTER INSERT ON sales
FOR EACH ROW
EXECUTE PROCEDURE tf_sales_bonus_ai();

CREATE OR REPLACE FUNCTION tf_sales_bonus_ad()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
  sub_amt NUMERIC(14, 2);
BEGIN
  sub_amt := ROUND(OLD.sale_price_rub * 0.05, 2);

  UPDATE realtor_bonuses b
  SET bonus_amount = GREATEST(b.bonus_amount - sub_amt, 0)
  WHERE b.realtor_id = OLD.realtor_id;

  RETURN OLD;
END;
$$;

CREATE TRIGGER trg_sales_bonus_ad
AFTER DELETE ON sales
FOR EACH ROW
EXECUTE PROCEDURE tf_sales_bonus_ad();
```

## Задание 2 (практика 3.2) — Сообщение при превышении максимального бонуса

**Условие задачи.** Создать триггер, который будет **информировать** о превышении размера накопленного бонуса риэлтором. **Максимальный размер бонуса установить самостоятельно.**

*Реализация в отчёте:* константа `max_bonus` в теле триггерной функции (в примере **500 000** руб.); при превышении — `RAISE NOTICE` (при необходимости замените порог или уровень сообщения).

```sql
CREATE OR REPLACE FUNCTION tf_realtor_bonuses_warn_max()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
  max_bonus CONSTANT NUMERIC(14, 2) := 500000;
BEGIN
  IF NEW.bonus_amount > max_bonus THEN
    RAISE NOTICE
      'Риэлтор id=%: накопленный бонус (%) превышает установленный максимум (%).',
      NEW.realtor_id,
      NEW.bonus_amount,
      max_bonus;
  END IF;

  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_realtor_bonuses_warn_max
AFTER INSERT OR UPDATE OF bonus_amount ON realtor_bonuses
FOR EACH ROW
EXECUTE PROCEDURE tf_realtor_bonuses_warn_max();
```

### Тест (задание 2, практика 3.2)

Вставка строки в `realtor_bonuses` с суммой **600 000** при пороге **500 000** в функции. Ожидается **`RAISE NOTICE`** о превышении максимума.

```sql
DO $$
DECLARE
  rid INTEGER;
BEGIN
  INSERT INTO realtors (last_name, passport_data)
  VALUES ('[Тест 3.2-2] Макс. бонус', '4500 123456')
  RETURNING id INTO rid;

  INSERT INTO realtor_bonuses (realtor_id, bonus_amount)
  VALUES (rid, 600000.00);
END $$;
```

## Задание 3 (практика 3.2) — Колонка паспортных данных и проверка по маске `XXXX YYYYYY`

**Условие задачи.** В таблицу «Риэлторы» добавить колонку **«Паспортные данные»**. Создать триггер, который будет проверять корректность паспортных данных по маске: **ХХХХ УУУУУУ**, где **Х** — серия паспорта, **У** — номер паспорта. **Между серией и номером паспорта пробел.** При несоответствии вводимой информации маске — **выводить сообщение**.

*Реализация в отчёте:* колонка `realtors.passport_data`; проверка регулярным выражением «4 цифры, пробел, 6 цифр»; при ошибке — `RAISE EXCEPTION` с текстом (операция вставки/обновления не выполняется).

```sql
CREATE OR REPLACE FUNCTION tf_realtors_passport_mask()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
  IF NEW.passport_data IS NULL OR BTRIM(NEW.passport_data) = '' THEN
    RETURN NEW;
  END IF;

  IF NEW.passport_data !~ '^\d{4} \d{6}$' THEN
    RAISE EXCEPTION
      'Паспортные данные должны соответствовать маске «XXXX YYYYYY»: 4 цифры серии, пробел, 6 цифр номера. Получено: %',
      NEW.passport_data
      USING ERRCODE = '23514';
  END IF;

  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_realtors_passport_mask
BEFORE INSERT OR UPDATE OF passport_data ON realtors
FOR EACH ROW
EXECUTE PROCEDURE tf_realtors_passport_mask();
```

### Тест (задание 3, практика 3.2)

Неверная строка паспорта — **`RAISE EXCEPTION`**, вставка не выполняется. Затем вставка с маской **4 цифры, пробел, 6 цифр** — успех.

```sql
DO $$
BEGIN
  BEGIN
    INSERT INTO realtors (last_name, passport_data)
    VALUES ('[Тест 3.2-3] Плохой паспорт', '1234567890');
  EXCEPTION
    WHEN OTHERS THEN
      RAISE NOTICE 'Ожидаемо: вставка отклонена — %', SQLERRM;
  END;
END $$;

INSERT INTO realtors (last_name, passport_data)
VALUES ('[Тест 3.2-3] Хороший паспорт', '4500 654321');

SELECT id, last_name, passport_data
FROM realtors
WHERE last_name LIKE '[Тест 3.2-3]%'
ORDER BY id;
```

## Задание 4 (практика 3.2) — Форматирование телефона `79996667788` → `+7 (999) 666 77 88`

**Условие задачи.** Создать триггер, который при добавлении нового риэлтора формат телефона **`79996667788`** преобразует в: **`+7 (999) 666 77 88`**.

*Реализация в отчёте:* колонка `realtors.phone`; триггер `BEFORE INSERT OR UPDATE OF phone` — из цифр строки собирается требуемый вид; также обрабатывается ввод из 10 цифр (без ведущей `7`).

```sql
CREATE OR REPLACE FUNCTION tf_realtors_phone_format()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
  d TEXT;
BEGIN
  IF NEW.phone IS NULL OR BTRIM(NEW.phone) = '' THEN
    RETURN NEW;
  END IF;

  d := REGEXP_REPLACE(NEW.phone, '\D', '', 'g');

  IF LENGTH(d) = 11 AND SUBSTRING(d FROM 1 FOR 1) = '7' THEN
    NEW.phone :=
      '+7 (' || SUBSTRING(d FROM 2 FOR 3) || ') '
      || SUBSTRING(d FROM 5 FOR 3) || ' '
      || SUBSTRING(d FROM 8 FOR 2) || ' '
      || SUBSTRING(d FROM 10 FOR 2);
  ELSIF LENGTH(d) = 10 THEN
    NEW.phone :=
      '+7 (' || SUBSTRING(d FROM 1 FOR 3) || ') '
      || SUBSTRING(d FROM 4 FOR 3) || ' '
      || SUBSTRING(d FROM 7 FOR 2) || ' '
      || SUBSTRING(d FROM 9 FOR 2);
  END IF;

  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_realtors_phone_format
BEFORE INSERT OR UPDATE OF phone ON realtors
FOR EACH ROW
EXECUTE PROCEDURE tf_realtors_phone_format();
```

### Тест (задание 4, практика 3.2)

Вставка риэлтора с «сырым» номером в формате из условия задачи. В таблице должно сохраниться **`+7 (999) 666 77 88`**.

```sql
INSERT INTO realtors (last_name, phone)
VALUES ('[Тест 3.2-4] Телефон', '79996667788')
RETURNING id, last_name, phone;
```
