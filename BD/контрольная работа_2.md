## Задание 1 — `products`: запрет цены `price <= 0` (BEFORE INSERT OR UPDATE)

```sql
DROP TRIGGER IF EXISTS trg_products_price_positive ON products;
DROP FUNCTION IF EXISTS trg_fn_products_price_positive();

CREATE OR REPLACE FUNCTION trg_fn_products_price_positive()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
  IF NEW.price <= 0 THEN
    RAISE EXCEPTION 'Цена товара должна быть положительной';
  END IF;
  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_products_price_positive
BEFORE INSERT OR UPDATE ON products
FOR EACH ROW
EXECUTE PROCEDURE trg_fn_products_price_positive();
```

Проверка:

```sql

INSERT INTO products (product_name, price) VALUES ('Тест', 10);

-- должно выбросить исключение
-- INSERT INTO products (product_name, price) VALUES ('Плохо', 0);
```

## Задание 2 — журнал изменения `salary` в `salary_log` (AFTER UPDATE OF salary)

```sql
-- важно: в базе может уже быть "чужая" таблица employees (например из рекурсии)
DROP TABLE IF EXISTS salary_log;
DROP TABLE IF EXISTS employees CASCADE;

CREATE TABLE employees(
  id SERIAL PRIMARY KEY,
  full_name TEXT,
  salary NUMERIC
);

CREATE TABLE salary_log(
  log_id SERIAL PRIMARY KEY,
  employee_id INTEGER,
  old_salary NUMERIC,
  new_salary NUMERIC,
  change_date TIMESTAMP
);

DROP TRIGGER IF EXISTS trg_employees_salary_log ON employees;
DROP FUNCTION IF EXISTS trg_fn_employees_salary_log();

CREATE OR REPLACE FUNCTION trg_fn_employees_salary_log()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
  IF OLD.salary IS DISTINCT FROM NEW.salary THEN
    INSERT INTO salary_log (employee_id, old_salary, new_salary, change_date)
    VALUES (NEW.id, OLD.salary, NEW.salary, NOW());
  END IF;
  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_employees_salary_log
AFTER UPDATE OF salary ON employees
FOR EACH ROW
EXECUTE PROCEDURE trg_fn_employees_salary_log();
```

Проверка:

```sql
INSERT INTO employees(full_name, salary) VALUES ('Иванов И.И.', 100000);

UPDATE employees SET salary = salary WHERE id = 1;        -- строка в лог не попадёт
UPDATE employees SET salary = salary * 1.1 WHERE id = 1;    -- одна запись в salary_log
SELECT * FROM salary_log ORDER BY log_id;
```

## Задание 3 — запрет удаления клиента при наличии заказов (BEFORE DELETE)

```sql
DROP TABLE IF EXISTS orders;
DROP TABLE IF EXISTS customers;

CREATE TABLE customers(
  id SERIAL PRIMARY KEY,
  full_name TEXT
);

CREATE TABLE orders(
  id SERIAL PRIMARY KEY,
  customer_id INTEGER REFERENCES customers(id)
);

DROP TRIGGER IF EXISTS trg_customers_no_delete_with_orders ON customers;
DROP FUNCTION IF EXISTS trg_fn_customers_no_delete_with_orders();

CREATE OR REPLACE FUNCTION trg_fn_customers_no_delete_with_orders()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
  IF EXISTS (SELECT 1 FROM orders WHERE customer_id = OLD.id) THEN
    RAISE EXCEPTION 'Невозможно удалить клиента с существующими заказами';
  END IF;
  RETURN OLD;
END;
$$;

CREATE TRIGGER trg_customers_no_delete_with_orders
BEFORE DELETE ON customers
FOR EACH ROW
EXECUTE PROCEDURE trg_fn_customers_no_delete_with_orders();
```

Проверка:

```sql
INSERT INTO customers(full_name) VALUES ('Петров П.П.');
INSERT INTO orders(customer_id) VALUES (1);

DELETE FROM orders WHERE customer_id = 1;
DELETE FROM customers WHERE id = 1;
```

## Задание 4 — `order_items`: `total = quantity * price`, без «ручного» `total` (BEFORE INSERT OR UPDATE)

Значение `total` всегда приводится к произведению; если в INSERT/UPDATE передали другое ненулевое несовпадающее значение — исключение (ручной подлог отличается от автозаполнения).

```sql
DROP TABLE IF EXISTS order_items;

CREATE TABLE order_items(
  id SERIAL PRIMARY KEY,
  order_id INTEGER,
  quantity INTEGER,
  price NUMERIC,
  total NUMERIC
);

DROP TRIGGER IF EXISTS trg_order_items_total_auto ON order_items;
DROP FUNCTION IF EXISTS trg_fn_order_items_total_auto();

CREATE OR REPLACE FUNCTION trg_fn_order_items_total_auto()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
  calc NUMERIC;
BEGIN
  calc := NEW.quantity * NEW.price;

  IF NEW.total IS NOT NULL AND NEW.total IS DISTINCT FROM calc THEN
    RAISE EXCEPTION
      'Поле total вычисляется автоматически как quantity * price; задавать его вручную нельзя';
  END IF;

  NEW.total := calc;
  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_order_items_total_auto
BEFORE INSERT OR UPDATE ON order_items
FOR EACH ROW
EXECUTE PROCEDURE trg_fn_order_items_total_auto();
```

Проверка:

```sql
INSERT INTO order_items (order_id, quantity, price, total)
VALUES (1, 2, 100, NULL);   -- total станет 200

INSERT INTO order_items (order_id, quantity, price, total)
VALUES (1, 2, 100, 200);    -- ок

-- INSERT INTO order_items (order_id, quantity, price, total)
-- VALUES (1, 2, 100, 999); -- исключение: total не равен 2 * 100
```

DROP TABLE IF EXISTS orders;
DROP TABLE IF EXISTS customers;
DROP TABLE IF EXISTS salary_log;

  id SERIAL PRIMARY KEY,

CREATE TABLE orders(

    RAISE EXCEPTION 'Невозможно удалить клиента с существующими заказами';

CREATE TRIGGER trg_customers_no_delete_with_orders

CREATE OR REPLACE FUNCTION trg_fn_order_items_total_auto()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
DECLARE calc NUMERIC;
BEGIN
  calc := NEW.quantity * NEW.price;
  IF NEW.total IS NOT NULL AND NEW.total IS DISTINCT FROM calc THEN
    RAISE EXCEPTION
