## Задание 1 — Табличная функция: студенты со средним баллом выше порога

```sql
DROP TABLE IF EXISTS grades;
DROP TABLE IF EXISTS students;

CREATE TABLE students(
  id SERIAL PRIMARY KEY,
  full_name TEXT
);

CREATE TABLE grades(
  id SERIAL PRIMARY KEY,
  student_id INTEGER REFERENCES students(id),
  subject TEXT,
  grade INTEGER
);

INSERT INTO students(full_name) VALUES
  ('Иванов Иван Иванович'),
  ('Петров Пётр Петрович'),
  ('Сидорова Анна Сергеевна');

INSERT INTO grades(student_id, subject, grade) VALUES
  (1, 'Математика', 5),
  (1, 'Физика',     4),
  (1, 'История',    5),
  (2, 'Математика', 3),
  (2, 'Физика',     4),
  (2, 'История',    3),
  (3, 'Математика', 5),
  (3, 'Физика',     5),
  (3, 'История',    4);

CREATE OR REPLACE FUNCTION get_students_above_avg(min_avg NUMERIC)
RETURNS TABLE(
  student_id INTEGER,
  full_name TEXT,
  avg_grade NUMERIC
)
LANGUAGE sql
AS $$
SELECT
  s.id AS student_id,
  s.full_name,
  AVG(g.grade)::NUMERIC AS avg_grade
FROM students s
JOIN grades g ON g.student_id = s.id
GROUP BY s.id, s.full_name
HAVING AVG(g.grade)::NUMERIC > min_avg
ORDER BY avg_grade DESC, s.id;
$$;
```

Пример вызова:

```sql
SELECT * FROM get_students_above_avg(4.0);
```

## Задание 2 — `get_subordinates_hierarchy`: все подчинённые (включая косвенных) через рекурсивный CTE

```sql
DROP TABLE IF EXISTS employees;

CREATE TABLE employees (
  id SERIAL PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  manager_id INTEGER REFERENCES employees(id)
);

INSERT INTO employees(name, manager_id) VALUES
  ('Директор', NULL),      -- id = 1
  ('Руководитель A', 1),   -- id = 2
  ('Руководитель B', 1),   -- id = 3
  ('Сотрудник A1', 2),     -- id = 4
  ('Сотрудник A2', 2),     -- id = 5
  ('Сотрудник B1', 3),     -- id = 6
  ('Стажёр A1-1', 4);      -- id = 7

CREATE OR REPLACE FUNCTION get_subordinates_hierarchy(p_employee_id INTEGER)
RETURNS TABLE(
  employee_id INTEGER,
  employee_name VARCHAR,
  level INTEGER
)
LANGUAGE sql
AS $$
WITH RECURSIVE hierarchy AS (
  -- уровень 0: сам сотрудник
  SELECT
    e.id   AS employee_id,
    e.name AS employee_name,
    0      AS level
  FROM employees e
  WHERE e.id = p_employee_id

  UNION ALL

  -- уровни 1..N: подчинённые
  SELECT
    sub.id   AS employee_id,
    sub.name AS employee_name,
    h.level + 1 AS level
  FROM employees sub
  JOIN hierarchy h ON sub.manager_id = h.employee_id
)
SELECT employee_id, employee_name, level
FROM hierarchy
ORDER BY level, employee_id;
$$;
```

Пример вызова:

```sql
SELECT * FROM get_subordinates_hierarchy(1);
```

## Задание 3 — Перебор `products`: увеличить цену каждого товара на 10% и вернуть число обработанных записей

```sql
DROP TABLE IF EXISTS products;

CREATE TABLE products(
  id SERIAL PRIMARY KEY,
  product_name TEXT,
  price NUMERIC
);

INSERT INTO products(product_name, price) VALUES
  ('Хлеб', 50),
  ('Молоко', 90.50),
  ('Сыр', 320.00),
  ('Акция (без цены)', NULL);

CREATE OR REPLACE FUNCTION increase_all_product_prices_10pct()
RETURNS INTEGER
LANGUAGE plpgsql
AS $$
DECLARE
  rec RECORD;
  processed_count INTEGER := 0;
BEGIN
  FOR rec IN
    SELECT id, price
    FROM products
    ORDER BY id
  LOOP
    processed_count := processed_count + 1;

    UPDATE products
      SET price = ROUND(price * 1.10, 2)
      WHERE id = rec.id
        AND price IS NOT NULL;
  END LOOP;

  RETURN processed_count;
END;
$$;
```

Пример вызова:

```sql
SELECT increase_all_product_prices_10pct() AS processed;
SELECT * FROM products ORDER BY id;
```
