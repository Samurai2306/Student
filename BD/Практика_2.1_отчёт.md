## Задание 1 — Функция для расчёта возраста

```sql
CREATE OR REPLACE FUNCTION calculate_age(birth_date DATE)
RETURNS INTEGER
LANGUAGE plpgsql
AS $$
BEGIN
  IF birth_date IS NULL THEN
    RETURN NULL;
  END IF;

  RETURN EXTRACT(YEAR FROM AGE(CURRENT_DATE, birth_date))::INT;
END;
$$;
```

## Задание 2 — Функция категоризации заказа

```sql
CREATE OR REPLACE FUNCTION categorize_order(order_amount DECIMAL(10,2))
RETURNS TEXT
LANGUAGE plpgsql
AS $$
BEGIN
  IF order_amount IS NULL THEN
    RETURN NULL;
  END IF;

  IF order_amount < 0 THEN
    RAISE EXCEPTION 'order_amount must be non-negative, got %', order_amount
      USING ERRCODE = '22003';
  END IF;

  IF order_amount < 1000 THEN
    RETURN 'МАЛЫЙ';
  ELSIF order_amount < 5000 THEN
    RETURN 'СРЕДНИЙ';
  ELSIF order_amount < 20000 THEN
    RETURN 'КРУПНЫЙ';
  ELSE
    RETURN 'ОСОБО КРУПНЫЙ';
  END IF;
END;
$$;
```

## Задание 3 — Функция генерации номеров заказов

```sql
CREATE OR REPLACE FUNCTION generate_order_numbers(count INTEGER)
RETURNS TABLE(order_number TEXT)
LANGUAGE plpgsql
AS $$
DECLARE
  i INTEGER;
  yyyymm TEXT;
  rnd TEXT;
BEGIN
  IF count IS NULL OR count <= 0 THEN
    RETURN;
  END IF;

  yyyymm := TO_CHAR(CURRENT_DATE, 'YYYYMM');

  FOR i IN 1..count LOOP
    rnd := LPAD((FLOOR(RANDOM() * 1000000))::INT::TEXT, 6, '0');
    order_number := 'ORD-' || yyyymm || rnd;
    RETURN NEXT;
  END LOOP;
END;
$$;
```

## Задание 4 — Функция перевода средств между счетами (транзакция и ошибки)

```sql
CREATE TABLE IF NOT EXISTS accounts (
  id SERIAL PRIMARY KEY,
  balance DECIMAL(10,2) NOT NULL DEFAULT 0
);

CREATE TABLE IF NOT EXISTS transactions (
  id SERIAL PRIMARY KEY,
  from_account INTEGER,
  to_account INTEGER,
  amount DECIMAL(10,2),
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE OR REPLACE FUNCTION transfer_money(
  from_account_id INTEGER,
  to_account_id INTEGER,
  amount DECIMAL(10,2)
)
RETURNS BOOLEAN
LANGUAGE plpgsql
AS $$
DECLARE
  from_balance DECIMAL(10,2);
BEGIN
  IF from_account_id IS NULL OR to_account_id IS NULL OR amount IS NULL THEN
    RETURN FALSE;
  END IF;

  IF amount <= 0 THEN
    RAISE EXCEPTION 'amount must be positive, got %', amount
      USING ERRCODE = '22003';
  END IF;

  IF from_account_id = to_account_id THEN
    RAISE EXCEPTION 'from_account_id and to_account_id must differ'
      USING ERRCODE = '22023';
  END IF;

  BEGIN
    SELECT a.balance
      INTO from_balance
      FROM accounts a
      WHERE a.id = from_account_id
      FOR UPDATE;

    IF NOT FOUND THEN
      RAISE EXCEPTION 'from account % not found', from_account_id
        USING ERRCODE = 'P0002';
    END IF;

    PERFORM 1
      FROM accounts a
      WHERE a.id = to_account_id
      FOR UPDATE;

    IF NOT FOUND THEN
      RAISE EXCEPTION 'to account % not found', to_account_id
        USING ERRCODE = 'P0002';
    END IF;

    IF from_balance < amount THEN
      RAISE EXCEPTION 'insufficient funds: balance %, requested %', from_balance, amount
        USING ERRCODE = 'P0001';
    END IF;

    UPDATE accounts
      SET balance = balance - amount
      WHERE id = from_account_id;

    UPDATE accounts
      SET balance = balance + amount
      WHERE id = to_account_id;

    INSERT INTO transactions(from_account, to_account, amount)
      VALUES (from_account_id, to_account_id, amount);

    RETURN TRUE;
  EXCEPTION
    WHEN OTHERS THEN
      RETURN FALSE;
  END;
END;
$$;
```
