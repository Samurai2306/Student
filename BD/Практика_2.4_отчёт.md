## Задание 2 — Функция: часы за неделю + «меньше/норма/больше»

```sql
CREATE OR REPLACE FUNCTION week_work_hours(p_employee_id INTEGER)
RETURNS TABLE(total_hours NUMERIC, status TEXT)
LANGUAGE sql
AS $$
WITH ordered AS (
  SELECT
    wd.employee_id,
    wd.event_ts,
    wd.reader_value,
    LEAD(wd.event_ts) OVER (PARTITION BY wd.employee_id ORDER BY wd.event_ts) AS next_ts,
    LEAD(wd.reader_value) OVER (PARTITION BY wd.employee_id ORDER BY wd.event_ts) AS next_val
  FROM рабочий_день wd
  WHERE wd.employee_id = p_employee_id
),
worked AS (
  SELECT
    SUM(
      CASE
        WHEN reader_value = 1 AND next_val = 2 AND next_ts > event_ts
          THEN EXTRACT(EPOCH FROM (next_ts - event_ts))
        ELSE 0
      END
    ) / 3600.0 AS hours
  FROM ordered
)
SELECT
  ROUND(COALESCE(hours, 0)::NUMERIC, 2) AS total_hours,
  CASE
    WHEN COALESCE(hours, 0) < 40 THEN 'Меньше нормы'
    WHEN COALESCE(hours, 0) = 40 THEN 'Норма'
    ELSE 'Больше нормы'
  END AS status
FROM worked;
$$;
```

## Задание 3 — Функция: расчёт зарплаты с учётом опозданий

```sql
CREATE OR REPLACE FUNCTION calc_salary_with_tardiness(
  p_employee_id INTEGER,
  base_salary NUMERIC
)
RETURNS NUMERIC
LANGUAGE sql
AS $$
WITH days AS (
  SELECT DISTINCT (event_ts::DATE) AS d
  FROM рабочий_день
  WHERE employee_id = p_employee_id
),
day_marks AS (
  SELECT
    d.d AS work_date,
    (SELECT MIN(event_ts) FROM рабочий_день WHERE employee_id = p_employee_id AND event_ts::DATE = d.d AND reader_value = 1) AS first_in,
    (SELECT MIN(event_ts) FROM рабочий_день WHERE employee_id = p_employee_id AND event_ts::DATE = d.d AND reader_value = 2 AND event_ts::TIME <= TIME '13:00:00') AS lunch_out,
    (SELECT MIN(event_ts) FROM рабочий_день WHERE employee_id = p_employee_id AND event_ts::DATE = d.d AND reader_value = 1 AND event_ts::TIME >= TIME '13:00:00') AS lunch_in,
    (SELECT MAX(event_ts) FROM рабочий_день WHERE employee_id = p_employee_id AND event_ts::DATE = d.d AND reader_value = 2) AS last_out
  FROM days d
),
late AS (
  SELECT
    SUM(
      GREATEST(0,
        EXTRACT(EPOCH FROM (COALESCE(first_in, work_date + TIME '09:00:00') - (work_date + TIME '09:00:00'))) / 60.0
      )
      +
      GREATEST(0,
        EXTRACT(EPOCH FROM ((work_date + TIME '13:00:00') - COALESCE(lunch_out, work_date + TIME '13:00:00'))) / 60.0
      )
      +
      GREATEST(0,
        EXTRACT(EPOCH FROM (COALESCE(lunch_in, work_date + TIME '14:00:00') - (work_date + TIME '14:00:00'))) / 60.0
      )
      +
      GREATEST(0,
        EXTRACT(EPOCH FROM ((work_date + TIME '18:00:00') - COALESCE(last_out, work_date + TIME '18:00:00'))) / 60.0
      )
    ) AS total_late_minutes
  FROM day_marks
),
coef AS (
  SELECT
    GREATEST(0, 1 - 0.05 * FLOOR(COALESCE(total_late_minutes, 0) / 10.0)) AS a
  FROM late
)
SELECT
  ROUND((base_salary + base_salary * a)::NUMERIC, 2)
FROM coef;
$$;
```
