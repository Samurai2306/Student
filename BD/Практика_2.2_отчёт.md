## Задание 1 — Конвертация стоимости объекта недвижимости в другую валюту

```sql
CREATE OR REPLACE FUNCTION convert_property_price(
  rate_rub NUMERIC,
  property_id INTEGER
)
RETURNS NUMERIC
LANGUAGE plpgsql
AS $$
DECLARE
  price_rub NUMERIC;
BEGIN
  IF rate_rub IS NULL OR property_id IS NULL THEN
    RETURN NULL;
  END IF;

  IF rate_rub <= 0 THEN
    RAISE EXCEPTION 'rate_rub must be > 0, got %', rate_rub
      USING ERRCODE = '22003';
  END IF;

  SELECT p.price_rub
    INTO price_rub
    FROM properties p
    WHERE p.id = property_id;

  IF NOT FOUND THEN
    RAISE EXCEPTION 'property % not found', property_id
      USING ERRCODE = 'P0002';
  END IF;

  RETURN price_rub / rate_rub;
END;
$$;
```

## Задание 2 — Зарплата риэлтора за месяц: \(N*S+R\)

```sql
CREATE OR REPLACE FUNCTION realtor_month_salary(
  s NUMERIC,
  r NUMERIC,
  month_num INTEGER,
  year_num INTEGER,
  realtor_last_name TEXT
)
RETURNS NUMERIC
LANGUAGE plpgsql
AS $$
DECLARE
  n_total NUMERIC;
BEGIN
  IF s IS NULL OR r IS NULL OR month_num IS NULL OR year_num IS NULL OR realtor_last_name IS NULL THEN
    RETURN NULL;
  END IF;

  IF month_num < 1 OR month_num > 12 THEN
    RAISE EXCEPTION 'month_num must be 1..12, got %', month_num
      USING ERRCODE = '22008';
  END IF;

  SELECT COALESCE(SUM(sa.sale_price_rub), 0)
    INTO n_total
    FROM sales sa
    JOIN realtors re ON re.id = sa.realtor_id
    WHERE re.last_name = realtor_last_name
      AND EXTRACT(YEAR FROM sa.sale_date)::INT = year_num
      AND EXTRACT(MONTH FROM sa.sale_date)::INT = month_num;

  RETURN n_total * s + r;
END;
$$;
```

## Задание 3 — Ежемесячный платёж по ипотеке (аннуитет)

```sql
CREATE OR REPLACE FUNCTION mortgage_monthly_payment(
  property_id INTEGER,
  annual_rate NUMERIC,
  years INTEGER,
  down_payment NUMERIC
)
RETURNS NUMERIC
LANGUAGE plpgsql
AS $$
DECLARE
  price_rub NUMERIC;
  principal NUMERIC;
  months INTEGER;
  r_month NUMERIC;
BEGIN
  IF property_id IS NULL OR annual_rate IS NULL OR years IS NULL OR down_payment IS NULL THEN
    RETURN NULL;
  END IF;

  IF years <= 0 THEN
    RAISE EXCEPTION 'years must be > 0, got %', years
      USING ERRCODE = '22003';
  END IF;

  IF annual_rate < 0 THEN
    RAISE EXCEPTION 'annual_rate must be >= 0, got %', annual_rate
      USING ERRCODE = '22003';
  END IF;

  SELECT p.price_rub
    INTO price_rub
    FROM properties p
    WHERE p.id = property_id;

  IF NOT FOUND THEN
    RAISE EXCEPTION 'property % not found', property_id
      USING ERRCODE = 'P0002';
  END IF;

  principal := price_rub - down_payment;
  IF principal <= 0 THEN
    RETURN 0;
  END IF;

  months := years * 12;
  r_month := annual_rate / 12 / 100;

  IF r_month = 0 THEN
    RETURN principal / months;
  END IF;

  RETURN principal * (r_month * POWER(1 + r_month, months)) / (POWER(1 + r_month, months) - 1);
END;
$$;
```

## Задание 4 — Список объектов: цена за 1м2 ниже средней по району

```sql
CREATE OR REPLACE FUNCTION properties_below_district_avg(
  p_address TEXT,
  p_district TEXT,
  p_rooms INTEGER
)
RETURNS TABLE(
  property_id INTEGER,
  address TEXT,
  district TEXT,
  rooms INTEGER,
  price_per_m2 NUMERIC,
  district_avg_price_per_m2 NUMERIC
)
LANGUAGE sql
AS $$
WITH base AS (
  SELECT
    p.id,
    p.address,
    p.district,
    p.rooms,
    (p.price_rub / NULLIF(p.area_m2, 0))::NUMERIC AS price_per_m2
  FROM properties p
  WHERE (p_address IS NULL OR p.address ILIKE '%' || p_address || '%')
    AND (p_district IS NULL OR p.district = p_district)
    AND (p_rooms IS NULL OR p.rooms = p_rooms)
),
avg_by_district AS (
  SELECT
    district,
    AVG(price_per_m2) AS avg_price_per_m2
  FROM base
  WHERE price_per_m2 IS NOT NULL
  GROUP BY district
)
SELECT
  b.id,
  b.address,
  b.district,
  b.rooms,
  b.price_per_m2,
  a.avg_price_per_m2
FROM base b
JOIN avg_by_district a ON a.district = b.district
WHERE b.price_per_m2 < a.avg_price_per_m2
ORDER BY b.district, b.price_per_m2 ASC;
$$;
```

## Задание 5 — Статистика по продажам за указанный год

```sql
CREATE OR REPLACE FUNCTION sales_stats_by_year(p_year INTEGER)
RETURNS TABLE(
  property_type TEXT,
  sales_count INTEGER,
  percent_of_total NUMERIC,
  total_sum NUMERIC
)
LANGUAGE sql
AS $$
WITH s AS (
  SELECT
    p.property_type::TEXT AS property_type,
    sa.sale_price_rub::NUMERIC AS sale_price_rub
  FROM sales sa
  JOIN properties p ON p.id = sa.property_id
  WHERE EXTRACT(YEAR FROM sa.sale_date)::INT = p_year
),
agg AS (
  SELECT
    property_type,
    COUNT(*)::INT AS sales_count,
    SUM(sale_price_rub)::NUMERIC AS total_sum
  FROM s
  GROUP BY property_type
),
tot AS (
  SELECT SUM(sales_count)::NUMERIC AS total_count
  FROM agg
)
SELECT
  a.property_type,
  a.sales_count,
  CASE
    WHEN t.total_count = 0 THEN 0
    ELSE ROUND((a.sales_count::NUMERIC / t.total_count) * 100, 2)
  END AS percent_of_total,
  a.total_sum
FROM agg a
CROSS JOIN tot t
ORDER BY a.sales_count DESC, a.property_type;
$$;
```
